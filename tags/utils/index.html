#时间戳转换为 YY-MM-DD hh:mm:ss 字符串格式

timeFormat(timestamp) {
//timestamp是整数，否则要parseInt转换
var time = new Date(timestamp);
var y = time.getFullYear();
var m = time.getMonth() + 1;
var d = time.getDate();
var h = time.getHours();
var mm = time.getMinutes();
